<Steps>
    <Step title="Installation">
    <CodeGroup>
        ```sh pnpm
        pnpm add -D sveltex-preprocess
        ```
        ```sh bun
        bun add -D sveltex-preprocess
        ```
        ```sh npm
        npm add -D sveltex-preprocess
        ```
        ```sh yarn
        yarn add -D sveltex-preprocess
        ```
    </CodeGroup>
    </Step>
    <Step title="Create configuration file">
        Create a `sveltex.config.js` file in the root of your project (or
        wherever, really, as long as you can import it to your
        `svelte.config.js`).

        ```js sveltex.config.js
        import { sveltex } from 'sveltex';

        export const sveltexPreprocessor = await sveltex(
            'none', // Markdown backend
            'none', // Code backend (syntax highlighting)
            'none', // TeX backend
            'none', // Advanced TeX backend
        );

        await sveltexPreprocessor.configure({
            // ...
        });
        ```
    </Step>
    <Step title="Add preprocessor to Svelte configuration">

        ```js svelte.config.js
        import { sveltexPreprocessor } from './sveltex.config.js';

        /** @type {import('@sveltejs/kit').Config} */
        const config = {
            preprocess: [sveltexPreprocessor],
        };

        export default config;
        ```
    </Step>

</Steps>
